Tricky Concepts

1. Mutable Default Arguments

The Problem:

def add_item(item, items=[]):
    items.append(item)
    return items

print(add_item(1))  # [1]
print(add_item(2))  # [1, 2] ← Unexpected!
print(add_item(3))  # [1, 2, 3] ← List persists!

Why? Default argument is evaluated once at function definition, not each call. The same list object is reused!

Solution:

def add_item(item, items=None):
    if items is None:
        items = []
    items.append(item)
    return items

print(add_item(1))  # [1]
print(add_item(2))  # [2] ← Correct!

Key Insight: Never use mutable objects (list, dict) as default arguments.

2. Late Binding Closures

The Problem:

functions = []
for i in range(3):
    functions.append(lambda: i)

for f in functions:
    print(f())  # 2, 2, 2 ← All print 2!

Why? Lambda captures variable i by reference, not value. When functions execute, i is already 2 (final value).

Solution:

# Method 1: Default argument
functions = [lambda i=i: i for i in range(3)]

# Method 2: functools.partial
from functools import partial
functions = [partial(lambda i: i, i) for i in range(3)]

for f in functions:
    print(f())  # 0, 1, 2 ← Correct!

3. Shallow vs Deep Copy

Shallow Copy:

import copy

original = [[1, 2], [3, 4]]
shallow = copy.copy(original)
# or: shallow = original.copy()
# or: shallow = list(original)

shallow[0][0] = 'X'
print(original)  # [['X', 2], [3, 4]] ← Modified!
print(shallow)   # [['X', 2], [3, 4]]

Deep Copy:

deep = copy.deepcopy(original)
deep[0][0] = 'Y'
print(original)  # [[1, 2], [3, 4]] ← Unchanged!
print(deep)      # [['Y', 2], [3, 4]]

Key Insight: Shallow copy copies references, deep copy copies recursively.

5. is vs ==

a = [1, 2, 3]
b = [1, 2, 3]
c = a

print(a == b)  # True (same values)
print(a is b)  # False (different objects)
print(a is c)  # True (same object)

# Special case: Small integers
x = 256
y = 256
print(x is y)  # True (cached)

z = 257
w = 257
print(z is w)  # False (not cached)

Rule: Use == for value comparison, is for identity.

6. Global Interpreter Lock (GIL)

Key Points:

  • Python has GIL: only one thread executes Python bytecode at a time
  • Affects CPU-bound multi-threaded programs
  • I/O-bound programs not affected
  • Solutions: multiprocessing, async/await

7. *args and **kwargs

def function(*args, **kwargs):
    print(f"args: {args}")      # Tuple
    print(f"kwargs: {kwargs}")  # Dict

function(1, 2, 3, a=4, b=5)
# args: (1, 2, 3)
# kwargs: {'a': 4, 'b': 5}

Unpacking:

def add(a, b, c):
    return a + b + c

args = [1, 2, 3]
print(add(*args))  # 6

kwargs = {'a': 1, 'b': 2, 'c': 3}
print(add(**kwargs))  # 6

Common Interview Questions

Q1: Reverse a String

Multiple approaches:

s = "hello"

# Method 1: Slicing (best)
reversed_s = s[::-1]  # "olleh"

# Method 2: reversed() function
reversed_s = ''.join(reversed(s))

# Method 3: Loop
reversed_s = ''
for char in s:
    reversed_s = char + reversed_s

# Method 4: List reverse
chars = list(s)
chars.reverse()
reversed_s = ''.join(chars)

Q2: Check Palindrome

def is_palindrome(s):
    # Clean string: remove spaces, lowercase
    s = s.replace(' ', '').lower()
    return s == s[::-1]

print(is_palindrome("A man a plan a canal Panama"))  # True
print(is_palindrome("race a car"))  # False

Q3: Find First Non-Repeating Character

def first_non_repeating(s):
    from collections import Counter
    
    counts = Counter(s)
    for char in s:
        if counts[char] == 1:
            return char
    return None

print(first_non_repeating("leetcode"))  # 'l'
print(first_non_repeating("loveleetcode"))  # 'v'

Q4: Two Sum Problem

def two_sum(nums, target):
    """
    Find indices of two numbers that add up to target.
    Time: O(n), Space: O(n)
    """
    seen = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i
    return None

print(two_sum([2, 7, 11, 15], 9))  # [0, 1]

Q5: FizzBuzz

def fizzbuzz(n):
    result = []
    for i in range(1, n + 1):
        if i % 15 == 0:
            result.append("FizzBuzz")
        elif i % 3 == 0:
            result.append("Fizz")
        elif i % 5 == 0:
            result.append("Buzz")
        else:
            result.append(str(i))
    return result

print(fizzbuzz(15))

Q6: Remove Duplicates from List

# Method 1: Set (doesn't preserve order)
nums = [1, 2, 2, 3, 4, 4, 5]
unique = list(set(nums))  # Order not guaranteed

# Method 2: Dict (preserves order, Python 3.7+)
unique = list(dict.fromkeys(nums))

# Method 3: Loop (preserves order)
unique = []
for num in nums:
    if num not in unique:
        unique.append(num)

# Method 4: List comprehension with seen set
seen = set()
unique = [x for x in nums if not (x in seen or seen.add(x))]

Q7: Flatten Nested List

def flatten(nested_list):
    result = []
    for item in nested_list:
        if isinstance(item, list):
            result.extend(flatten(item))  # Recursive
        else:
            result.append(item)
    return result

nested = [1, [2, 3, [4, 5]], 6, [7, [8, 9]]]
print(flatten(nested))  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

Q8: Find Missing Number

def find_missing(nums):
    """
    Array contains 0 to n with one number missing.
    Time: O(n), Space: O(1)
    """
    n = len(nums)
    expected_sum = n * (n + 1) // 2
    actual_sum = sum(nums)
    return expected_sum - actual_sum

print(find_missing([0, 1, 3]))  # 2
print(find_missing([0, 1, 2, 3, 4, 6]))  # 5

Q9: Valid Parentheses

def is_valid_parentheses(s):
    stack = []
    mapping = {')': '(', '}': '{', ']': '['}
    
    for char in s:
        if char in mapping:
            top = stack.pop() if stack else '#'
            if mapping[char] != top:
                return False
        else:
            stack.append(char)
    
    return not stack

print(is_valid_parentheses("()"))  # True
print(is_valid_parentheses("()[]{}"))  # True
print(is_valid_parentheses("(]"))  # False

Q10: Anagram Check

def is_anagram(s1, s2):
    # Method 1: Sorted
    return sorted(s1) == sorted(s2)
    
    # Method 2: Counter
    from collections import Counter
    return Counter(s1) == Counter(s2)

print(is_anagram("listen", "silent"))  # True
print(is_anagram("hello", "world"))  # False

Best Practices

1. PEP 8 Style Guide

# Naming conventions
MY_CONSTANT = 100           # Constants: UPPER_CASE
my_variable = 10            # Variables: snake_case
my_function()               # Functions: snake_case

class MyClass:              # Classes: PascalCase
    def my_method(self):    # Methods: snake_case
        pass

# Indentation: 4 spaces
def function():
    if condition:
        do_something()
    else:
        do_other()

# Line length: 79 characters
long_variable = (some_long_expression +
                 another_expression)

# Imports
import os
import sys

from collections import Counter  # One per line

2. Pythonic Code

# List comprehension (good)
squares = [x**2 for x in range(10)]

# Instead of loop (verbose)
squares = []
for x in range(10):
    squares.append(x**2)

# Dictionary comprehension
square_dict = {x: x**2 for x in range(5)}

# Enumerate instead of range(len())
for i, item in enumerate(items):
    print(f"{i}: {item}")

# zip for parallel iteration
for name, age in zip(names, ages):
    print(f"{name} is {age}")

# any/all for boolean checks
if any(x > 10 for x in numbers):
    print("At least one > 10")

if all(x > 0 for x in numbers):
    print("All positive")

3. Context Managers

# Always use with for files
with open('file.txt', 'r') as f:
    content = f.read()
# File automatically closed

# Custom context manager
from contextlib import contextmanager

@contextmanager
def timer():
    import time
    start = time.time()
    yield
    end = time.time()
    print(f"Time: {end - start:.2f}s")

with timer():
    # Code to time
    sum(range(1000000))

finally:
    cleanup()

# EAFP vs LBYL
# EAFP (Easier to Ask Forgiveness than Permission)
try:
    value = my_dict[key]
except KeyError:
    value = default

# LBYL (Look Before You Leap)
if key in my_dict:
    value = my_dict[key]
else:
    value = default

# EAFP is more Pythonic

Performance Optimization

1. Use Built-in Functions

# Slow
total = 0
for num in numbers:
    total += num

# Fast
total = sum(numbers)

2. List Comprehension vs Loop

import timeit

# Comprehension (faster)
result = [x**2 for x in range(1000)]

# Loop (slower)
result = []
for x in range(1000):
    result.append(x**2)

3. Set for Membership Testing

# Slow O(n)
if item in my_list:
    pass

# Fast O(1)
if item in my_set:
    pass

4. Join vs Concatenation

# Slow
result = ''
for s in strings:
    result += s

# Fast
result = ''.join(strings)
